math方法实例：
print(math.pi)
3.141592653589793
print(math.e)
2.718281828459045
print(math.fabs(-10))#取绝对值，返回float类型。基于模块math函数

print(math.ceil(4.01))#向上取整
5
print(math.floor(4.01))#向下取整

print(math.exp(1))#e的n次幂
2.718281828459045

非math方法实例：
print(abs(-10))#取绝对值。不是基于模块math的

print(max(1,2,3,4,5,6))#最大值
6
list = [1,2,3,45,56]
print(max(list))#最大值
56
print(min(1,3,4,5,6))#最小值
1
print(round(4.4))#四舍五入

print(round(4.6454546,2))#取小数点几位四舍五入
4.65





#随机函数random
import random
print(dir(random))
['BPF', 'LOG4', 'NV_MAGICCONST', 'RECIP_BPF', 'Random', 'SG_MAGICCONST', 'SystemRandom', 'TWOPI', '_BuiltinMethodType', '_MethodType', '_Sequence', '_Set', '__all__', '__builtins__', '__cached__', '__doc__', '__file__', '__loader__', '__name__', '__package__', '__spec__', '_acos', '_bisect', '_ceil', '_cos', '_e', '_exp', '_inst', '_itertools', '_log', '_pi', '_random', '_sha512', '_sin', '_sqrt', '_test', '_test_generator', '_urandom', '_warn', 'betavariate', 'choice', 'choices', 'expovariate', 'gammavariate', 'gauss', 'getrandbits', 'getstate', 'lognormvariate', 'normalvariate', 'paretovariate', 'randint', 'random', 'randrange', 'sample', 'seed', 'setstate', 'shuffle', 'triangular', 'uniform', 'vonmisesvariate', 'weibullvariate']

#random.choice(seq)#从序列中随机挑选一个整数
print(random.choice(range(10)))
9
#randrange([start,]stop[,step])#从范围内随机获得随机数
print(random.randrange(0,10,3))
6
#random()#随机生成下一个实数，范围在[0-1]内
print(random.random())
0.7607425209016444
#seed([x])#改变随机数生成的种子seed
print(random.seed(5))
None
#shuffle(lst)#随机排序
list = [1,2,0,3,4,5]
print(random.shuffle(list))#将序列的所有元素随机排序
print(list)
None
[2, 1, 3, 5, 0, 4]
#uniform(x,y)#均匀分布
print(random.uniform(5,6))#随机生成下一个实数，范围在[x,y]内
5.371793355562307
print(random.uniform(5,6))
5.8684454578650955
